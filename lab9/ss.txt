ALTER TABLE address IF EXISTS public.address ADD COLUMN IF NOT EXISTS latitude float8;
ALTER TABLE address IF EXISTS public.address ADD COLUMN IF NOT EXISTS longitude float8;
DROP FUNCTION filter_addresses();
CREATE OR REPLACE FUNCTION filter_addresses()
RETURNS TABLE (
		address character varying(50),
		address_id int
)
LANGUAGE plpgsql
AS $$
DECLARE 
BEGIN
return query
select A.address, A.address_id from address A
	where A.city_id > 400 AND A.city_id < 600 and A.address like '%11%';
END;
$$
